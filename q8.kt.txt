// 8. Write a program to count the number of digits in a given integer.
fun main() {
    var number = 987654 // Example input
    var count = 0
    val temp = number

    // Keep dividing by 10 until the number becomes 0
    while (number != 0) {
        number /= 10
        count++
    }

    println("The number of digits in $temp is: $count")
}
